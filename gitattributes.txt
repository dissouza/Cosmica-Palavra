# ----------------------------------------------------
# 1. Ignorar Arquivos/Diretórios do WordPress (Vendored)
# ----------------------------------------------------

# Ignora o diretório 'wp-admin' (código do core do WP)
wp-admin/** linguist-vendored

# Ignora o diretório 'wp-includes' (código do core do WP)
wp-includes/** linguist-vendored

# Ignora plugins e temas inteiros que são de terceiros (ex: woocommerce)
# Se você tiver um plugin ou tema que NAO É seu, use essa regra:
# wp-content/plugins/nome-do-plugin/** linguist-vendored
# wp-content/themes/nome-do-tema/** linguist-vendored

# Se o seu tema ou plugin é o código principal, NÃO use a regra acima.
# Use as regras abaixo para ignorar arquivos comuns de terceiros em temas/plugins:

# Ignora arquivos minificados de CSS e JS (comuns em libs ou assets de terceiros)
*.min.css linguist-vendored
*.min.js linguist-vendored

# Ignora arquivos de biblioteca dentro de diretórios comuns de terceiros
vendor/** linguist-vendored
assets/vendor/** linguist-vendored
node_modules/** linguist-vendored

# ----------------------------------------------------
# 2. Garantir Detecção das Suas Linguagens Principais
# ----------------------------------------------------

# O Linguist geralmente detecta .html, .css, .js e .php corretamente.
# As regras a seguir são mais para garantir ou para casos de arquivos sem extensão.

# Força arquivos .php a serem PHP (raramente necessário, mas garante)
*.php linguist-language=PHP

# Força arquivos .js a serem JavaScript
*.js linguist-language=JavaScript
